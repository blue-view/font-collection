    <template>
  <div>
    <el-table v-bind="$attrs" v-on="$listeners">
      <template v-for="(col, index) in columns">
        <!-- 操作列/自定义列 -->
        <slot v-if="col.slot" :name="col.slot"></slot>
        <!-- 普通列 -->
        <el-table-column
          v-else
          :key="index"
          :prop="col.prop"
          :label="col.label"
          :width="col.width"
          :min-width="col.minWidth"
          :formatter="col.formatter"
          :sortable="col.sortable"
        >
        </el-table-column>
      </template>
    </el-table>
    <Pagination
      v-bind:child-msg="pageparm"
      @callFather="callFather"
      v-if="isShowPage"
    ></Pagination>
  </div>
</template>

<script>
export default {
  name: "CustomTable",
  props: {
    columns: {
      type: Array,
      default: () => {
        return [];
      },
    },
    isShowPage: {
      type: Boolean,
      default: () => {
        return false;
      },
    },
    pageparm: {
      type: Object,
      default: () => {
        return {
          currentPage: 1,
          pageSize: 10,
          total: 10,
        };
      },
    },
  },
  methods: {
    callFather(param) {
      this.$emit("callFather", param);
    },
  },
};
</script>
