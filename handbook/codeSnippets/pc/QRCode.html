    # source

    <template>
  <body>
    <slot></slot>
    <div id="qrCodeContainer">
      <div id="qrCode" ref="qrCodeDiv"></div>
    </div>
  </body>
</template>
<script>
import QRCode from "qrcodejs2";

export default {
  name: "CustomQRCode",
  data() {
    return {};
  },
  props: {
    colorDark: {
      type: String,
      default: "#333333",
    },
    colorLight: {
      type: String,
      default: "#ffffff",
    },
  },
  mounted: function () {},
  methods: {
    bindQRCode(QRCodeText) {
      this.$refs.qrCodeDiv.innerHTML = "";
      new QRCode(this.$refs.qrCodeDiv, {
        text: QRCodeText,
        width: 300,
        height: 300,
        colorDark: this.colorDark, //二维码颜色
        colorLight: this.colorLight, //二维码背景色
        correctLevel: QRCode.CorrectLevel.L, //容错率，L/M/H
      });
    },
  },
};
</script>

<style lang='less'>

}
